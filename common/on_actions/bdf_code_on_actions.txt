# For on_actions launched by the code itself.

# Root = Country
on_yearly_pulse_country = {
	on_actions = {
		bdf_on_yearly_events
	}
}

bdf_on_yearly_events = {
	effect = {
		trigger_event = { # Colonial / Imperial Conference
			id = bdf.10
			days = { 60 240 }
		}
		trigger_event = { # Colonial / Imperial Conference (hidden) --> Still fires but player doesn't get an event
			id = bdf.11
			days = { 60 240 }
		}
	}
}

# Root = Releasing Country
# scope:target = Released Country
on_country_released_as_own_subject = {
	on_actions = {
		bdf_on_country_released_as_own_subject
	}
}

bdf_on_country_released_as_own_subject = {
	effect = {
		if = {
			limit = {
				exists = c:GBR
				root = c:GBR
			}
			scope:target = {
				trigger_event = {
					id = bdf.1
				}
			}
		}
	}
}
